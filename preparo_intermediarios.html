// Adicione ao doPost existente, dentro do try:

// ── Buscar funcionários ──
if (data.tipo === 'funcionarios') {
  var ss = SpreadsheetApp.openById(data.sheetId);
  var sheet = ss.getSheetByName('Funcionárioss');
  var vals = sheet.getRange('A1:A50').getValues();
  var lista = vals.map(r => r[0]).filter(v => v && v !== '');
  return ok({funcionarios: lista});
}

// ── Salvar Limpeza ──
if (data.tipo === 'limpeza') {
  var ss = SpreadsheetApp.openById(data.sheetId);
  var sheet = ss.getSheetByName('Limpeza');
  var id = sheet.getLastRow() < 1 ? 1 : sheet.getLastRow();
  sheet.appendRow([id, data.qrCode, data.qtdInicial,
    data.horaInicio, data.qtdFinal, data.horaFim, data.funcionarios]);
  return ok({id: id});
}

// ── Salvar Cozinha ──
if (data.tipo === 'cozinha') {
  var ss = SpreadsheetApp.openById(data.sheetId);
  var sheet = ss.getSheetByName('Cozinha');
  var id = sheet.getLastRow() < 1 ? 1 : sheet.getLastRow();
  sheet.appendRow([id, data.qrCode, data.horaInicio,
    data.temperatura, data.tempo, data.horaFim]);
  return ok({id: id});
}

// ── Salvar Trituração ──
if (data.tipo === 'trituração') {
  var ss = SpreadsheetApp.openById(data.sheetId);
  var sheet = ss.getSheetByName('Trituração');
  var id = sheet.getLastRow() < 1 ? 1 : sheet.getLastRow();
  var qrNovo = 'MB-' + String(id).padStart(5,'0');
  sheet.appendRow([id, data.qrCode, data.horaInicio,
    data.horaFim, data.kgMassa, qrNovo]);
  return ok({id: id, qrNovo: qrNovo});
}

// ── Salvar Fatiamento ──
if (data.tipo === 'fatiamento') {
  var ss = SpreadsheetApp.openById(data.sheetId);
  var sheet = ss.getSheetByName('Fatiamento e Bandejamento');
  var id = sheet.getLastRow() < 1 ? 1 : sheet.getLastRow();
  var qrNovo = 'FT-' + String(id).padStart(5,'0');
  sheet.appendRow([id, data.qrCode, data.quantidade,
    data.horaInicio, data.horaFim, qrNovo]);
  return ok({id: id, qrNovo: qrNovo});
}
