function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.openById(data.sheetId);
    var sheet = ss.getSheetByName('Entrada MPrima');
    
    var lastRow = sheet.getLastRow();
    var newId = lastRow < 1 ? 1 : lastRow; // ID sequencial
    var qrCode = 'MP-' + String(newId).padStart(5, '0');
    
    sheet.appendRow([
      newId,
      data.nota_fiscal,
      data.laudo,
      data.dt_entrada,
      data.fornecedor,
      data.marca,
      data.validade,
      data.produto,
      data.unidade,
      data.quantidade,
      qrCode
    ]);
    
    return ContentService
      .createTextOutput(JSON.stringify({
        success: true,
        id: newId,
        qrCode: qrCode
      }))
      .setMimeType(ContentService.MimeType.JSON);
      
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({success: false, error: err.message}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  return ContentService
    .createTextOutput(JSON.stringify({status: 'online'}))
    .setMimeType(ContentService.MimeType.JSON);
}
