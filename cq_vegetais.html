function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.openById(data.sheetId);

    // ── Entrada MPrima (já existente) ──
    if (data.tipo === 'entrada') {
      var sheet = ss.getSheetByName('Entrada MPrima');
      var lastRow = sheet.getLastRow();
      var newId = lastRow < 1 ? 1 : lastRow;
      var qrCode = 'MP-' + String(newId).padStart(5, '0');
      sheet.appendRow([newId, data.nota_fiscal, data.laudo,
        data.dt_entrada, data.fornecedor, data.marca,
        data.validade, data.produto, data.unidade,
        data.quantidade, qrCode]);
      return ok({id: newId, qrCode: qrCode});
    }

    // ── Buscar entrada pelo QR ──
    if (data.tipo === 'buscar') {
      var sheet = ss.getSheetByName('Entrada MPrima');
      var values = sheet.getDataRange().getValues();
      for (var i = 1; i < values.length; i++) {
        if (values[i][10] === data.qrCode) {
          return ok({
            encontrado: true,
            id: values[i][0],
            nota_fiscal: values[i][1],
            laudo: values[i][2],
            dt_entrada: values[i][3],
            fornecedor: values[i][4],
            marca: values[i][5],
            validade: values[i][6],
            produto: values[i][7],
            unidade: values[i][8],
            quantidade: values[i][9],
            qrCode: values[i][10]
          });
        }
      }
      return ok({encontrado: false});
    }

    // ── Salvar CQ ──
    if (data.tipo === 'cq') {
      var sheetCQ = ss.getSheetByName('CQ vegetais');
      var lastCQ = sheetCQ.getLastRow();
      var newCQId = lastCQ < 1 ? 1 : lastCQ;

      // Upload foto para Drive
      var fotoLink = '';
      if (data.fotoBase64 && data.fotoNome) {
        var folder = DriveApp.getFolderById(data.folderId);
        var blob = Utilities.newBlob(
          Utilities.base64Decode(data.fotoBase64),
          data.fotoMime,
          data.fotoNome
        );
        var file = folder.createFile(blob);
        file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
        fotoLink = file.getUrl();
      }

      sheetCQ.appendRow([
        newCQId,
        data.dt_entrada,
        data.qrCode,
        data.visual,
        fotoLink,
        data.espessura,
        data.umidade,
        data.metais
      ]);
      return ok({id: newCQId, fotoLink: fotoLink});
    }

  } catch(err) {
    return erro(err.message);
  }
}

function ok(data) {
  return ContentService
    .createTextOutput(JSON.stringify({success: true, ...data}))
    .setMimeType(ContentService.MimeType.JSON);
}

function erro(msg) {
  return ContentService
    .createTextOutput(JSON.stringify({success: false, error: msg}))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  return ContentService
    .createTextOutput(JSON.stringify({status: 'online'}))
    .setMimeType(ContentService.MimeType.JSON);
}
